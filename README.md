local Toggled = false
game:GetService("UserInputService").InputBegan:connect(function(Key, g)
	if g then return end;
	if Key.KeyCode == Enum.KeyCode.V then
		if Toggled == false then
			Toggled = true
			Main.Visible = false
		elseif Toggled == true then
			Toggled = false;
			Main.Visible = true;
		end;
	end;
end)'
